jQueryNiftyBorders demo page

back to the home Fork on github

Demo 1

Code

$("#demo1 a img").niftyBorders({
    borderColor: "#dfdfdf",
    borderTickness: 10,
    borderHoverTickness : 0
});

Demo 2

Code

$("#demo2 a img").niftyBorders({
	borderColor: "yellow",
	borderTickness: 0,
	borderHoverTickness : 20,
	borderOpacity: .0,
	borderHoverOpacity: 1
});

Demo 3

Code

$("#demo3 a img").niftyBorders({
	speed: 500,
	easing: "linear",
	borderColor: "#cc4466",
	borderOpacity: .5,
	borderTickness: 10,
	borderHoverTickness : 20
});

Demo 4

Code

$("#demo4 a img").niftyBorders({
	borderColor: "cyan",
	borderType: "double",
	borderTickness: 10,
	borderHoverTickness : 0
});

Demo 5

Code

$("#demo5 a img").niftyBorders({
	borderColor: "#aaa",
	onMouseEnter: function(){ $("#status").html("mouseEnter!") },
	onMouseLeave: function(){ $("#status").html("mouseLeave!") }
});
